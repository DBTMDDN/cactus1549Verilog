//Wire
module top_module(
input in,
output out );
assign out=in;
endmodule
//Four Wire
module top_module(
input a,
input b,
input c,
output w,
output x,
output y,
output z );
assign w=a;
assign x=b;
assign y=b;
assign z=c;
endmodule
//Inverter
module top_module(
input in,
output out );
assign out=~in;
endmodule
//AND
module top_module(
input a,
input b,
output out );
assign out=a&b;

endmodule
//NOR
module top_module(
input a,
input b,
output out );
assign out=~(a|b);
endmodule
//XNOR
module top_module(
input a,
input b,
output out );
assign out=~(a^b);
endmodule

`default_nettype none
module top_module(
input a,
input b,
input c,
input d,
output out,
output out_n );
assign out=(a&b)|(c&d);
assign out_n=!((a&b)|(c&d));
endmodule
